function doPost(e) { var data = JSON.parse(e.postData.contents); var day = data.day || "Workout"; var sets = data.sets || []; var html = '<div style="font-family:sans-serif;max-width:480px">'; html += '<h2 style="color:#1a1a1a">💪 ' + day + ' Complete — ' + new Date().toDateString() + '</h2>'; html += '<table style="width:100%;border-collapse:collapse">'; sets.forEach(function(s) { html += '<tr style="border-bottom:1px solid #f0f0f0">'; html += '<td style="padding:8px 0">' + s.name + '</td>'; html += '<td style="padding:8px 0;color:#666">' + s.totalSets + '</td>'; html += '<td align="right" style="padding:8px 0;font-weight:600">' + s.topSet; if (s.pr) html += ' 🔥'; html += '</td></tr>'; }); html += '</table>'; html += '<p style="margin-top:16px;color:#666">Streak: <strong>' + (data.streak||0) + ' days 🔥</strong></p>'; html += '</div>'; GmailApp.sendEmail(Session.getActiveUser().getEmail(), "💪 " + day + " Complete — " + new Date().toDateString(), day + " complete.", { htmlBody: html }); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); }
